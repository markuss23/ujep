data_h <- read.csv("/home/marek/projects/ujep/zzd/zapocet/01/202405.csv")

data_h$date <- ymd_hms(data_h$date)
data_h$hour <- hour(data_h$date) # Pouit funkce hour() z lubridate k extrakci hodin

top5_crime_labels <- counts$types
filtered_data <- data_h %>%
  filter(types %in% top5_crime_labels)

average_crime <- filtered_data %>%
  group_by(types, hour) %>%
  summarize(avg_reports = mean(types)) # Opraven chybjcho zvrenho symbolu
## `summarise()` has grouped output by 'types'. You can override using the
## `.groups` argument.
# Vykreslen sloupcovch graf pro kadou podkategorii trestnch in
for (crime_label in unique(average_crime$types)) {
  crime_data <- filter(average_crime, types == crime_label)
  p <- ggplot(crime_data, aes(x = hour, y = avg_reports, fill = types)) +
    geom_bar(stat = "identity") +
    labs(title = crime_label, x = "Hodina", y = "Prmrn poet hlen", fill = "Hlavn kategorie") +
    theme_minimal()
  print(p)
}
